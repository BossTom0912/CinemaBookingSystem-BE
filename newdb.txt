  ### 1. Các cột được bổ sung vào các bảng hiện có:

  Bảng  USER :

  •  SpamViolationCount  (Số lần vi phạm spam)
  •  IsBlocked  (Trạng thái bị khóa)
  •  BlockedUntil  (Thời hạn mở khóa)

  Bảng  MOVIE :

  •  ViewCount  (Lượt xem)
  •  AverageRating  (Điểm đánh giá trung bình)
  •  TotalReviews  (Tổng số lượt đánh giá)
  •  TotalViews  (Tổng số lượt truy cập/xem)
  •  DailyViews  (Lượt xem trong ngày)

  Bảng  REVIEW :

  •  Status  (Trạng thái đánh giá: Phê duyệt, Từ chối, Đánh dấu, ...)
  •  EditCount  (Số lần đã chỉnh sửa bình luận)
  •  RejectedReason  (Lý do từ chối nếu bị AI đánh dấu)
  •  ModeratedBy  (Người hoặc AI đã kiểm duyệt)
  ──────
  ### 2. Các bảng (Tables/Entities) hoàn toàn mới được thêm vào:

  •  ChatHistory : Bảng lưu trữ lịch sử trò chuyện của chatbot AI.
  •  MovieViewLog : Bảng ghi lại chi tiết các lượt xem của mỗi phim.
  •  MovieDailyView : Bảng tổng hợp số lượng lượt xem của phim theo từng ngày.
  •  ReviewEditHistory : Bảng lưu lại các lần chỉnh sửa bình luận để đối chiếu.
  •  ReviewModerationHistory : Bảng ghi lại lịch sử kiểm duyệt (Moderation) các bài đánh giá.

=========================================
CÁC CÂU LỆNH SQL TƯƠNG ỨNG ĐƯỢC TẠO RA:
=========================================

-- 1. Thêm cột vào bảng USER
ALTER TABLE [USER] ADD [spamViolationCount] INT NOT NULL DEFAULT 0;
ALTER TABLE [USER] ADD [isBlocked] BIT NOT NULL DEFAULT 0;
ALTER TABLE [USER] ADD [blockedUntil] DATETIME2 NULL;

-- 2. Thêm cột vào bảng MOVIE
ALTER TABLE [MOVIE] ADD [viewCount] INT NOT NULL DEFAULT 0;
ALTER TABLE [MOVIE] ADD [averageRating] DECIMAL(3,2) NOT NULL DEFAULT 0.00;
ALTER TABLE [MOVIE] ADD [totalReviews] INT NOT NULL DEFAULT 0;
ALTER TABLE [MOVIE] ADD [totalViews] INT NOT NULL DEFAULT 0;
ALTER TABLE [MOVIE] ADD [dailyViews] INT NOT NULL DEFAULT 0;

-- 3. Thêm cột vào bảng REVIEW
ALTER TABLE [REVIEW] ADD [editCount] INT NOT NULL DEFAULT 0;
ALTER TABLE [REVIEW] ADD [rejectedReason] NVARCHAR(1000) NULL;
ALTER TABLE [REVIEW] ADD [moderatedBy] NVARCHAR(50) NULL;
ALTER TABLE [REVIEW] ADD CONSTRAINT [FK_REVIEW_MODERATED_BY] FOREIGN KEY ([moderatedBy]) REFERENCES [USER]([userId]);

-- 4. Tạo các bảng mới
CREATE TABLE [CHAT_HISTORY] (
    [chatHistoryId] NVARCHAR(50) PRIMARY KEY,
    [userId] NVARCHAR(50) NULL,
    [sessionId] NVARCHAR(100) NULL,
    [message] NVARCHAR(MAX) NOT NULL,
    [isUserMessage] BIT NOT NULL DEFAULT 1,
    [createdAt] DATETIME2 NOT NULL DEFAULT SYSUTCDATETIME(),
    CONSTRAINT [FK_CHAT_HISTORY_USER] FOREIGN KEY ([userId]) REFERENCES [USER]([userId])
);

CREATE TABLE [MOVIE_VIEW_LOG] (
    [viewLogId] NVARCHAR(50) PRIMARY KEY,
    [movieId] NVARCHAR(50) NOT NULL,
    [userId] NVARCHAR(50) NULL,
    [viewTime] DATETIME2 NOT NULL DEFAULT SYSUTCDATETIME(),
    [ipAddress] NVARCHAR(100) NULL,
    CONSTRAINT [FK_MOVIE_VIEW_LOG_MOVIE] FOREIGN KEY ([movieId]) REFERENCES [MOVIE]([movieId]),
    CONSTRAINT [FK_MOVIE_VIEW_LOG_USER] FOREIGN KEY ([userId]) REFERENCES [USER]([userId])
);

CREATE TABLE [MOVIE_DAILY_VIEW] (
    [dailyViewId] NVARCHAR(50) PRIMARY KEY,
    [movieId] NVARCHAR(50) NOT NULL,
    [viewDate] DATE NOT NULL,
    [viewCount] INT NOT NULL DEFAULT 0,
    CONSTRAINT [UQ_MOVIE_DAILY_VIEW] UNIQUE ([movieId], [viewDate]),
    CONSTRAINT [FK_MOVIE_DAILY_VIEW_MOVIE] FOREIGN KEY ([movieId]) REFERENCES [MOVIE]([movieId])
);

CREATE TABLE [REVIEW_EDIT_HISTORY] (
    [editHistoryId] NVARCHAR(50) PRIMARY KEY,
    [reviewId] NVARCHAR(50) NOT NULL,
    [oldComment] NVARCHAR(1000) NULL,
    [newComment] NVARCHAR(1000) NULL,
    [oldRating] INT NULL,
    [newRating] INT NULL,
    [editedAt] DATETIME2 NOT NULL DEFAULT SYSUTCDATETIME(),
    CONSTRAINT [FK_REVIEW_EDIT_HISTORY_REVIEW] FOREIGN KEY ([reviewId]) REFERENCES [REVIEW]([reviewId])
);

CREATE TABLE [REVIEW_MODERATION_HISTORY] (
    [moderationHistoryId] NVARCHAR(50) PRIMARY KEY,
    [reviewId] NVARCHAR(50) NOT NULL,
    [moderatedBy] NVARCHAR(50) NOT NULL,
    [oldStatus] VARCHAR(20) NULL,
    [newStatus] VARCHAR(20) NOT NULL,
    [reason] NVARCHAR(1000) NULL,
    [moderatedAt] DATETIME2 NOT NULL DEFAULT SYSUTCDATETIME(),
    CONSTRAINT [FK_REVIEW_MODERATION_HISTORY_REVIEW] FOREIGN KEY ([reviewId]) REFERENCES [REVIEW]([reviewId]),
    CONSTRAINT [FK_REVIEW_MODERATION_HISTORY_USER] FOREIGN KEY ([moderatedBy]) REFERENCES [USER]([userId])
);
