SELECT
    u.[email],
    N'Test@123' AS [password],
    r.[roleName],
    u.[status],
    u.[emailVerified]
FROM [USER] u
JOIN [ROLE] r ON r.[roleId] = u.[roleId]
WHERE u.[email] IN (
    N'customer@test.com',
    N'staff@test.com',
    N'manager@test.com',
    N'admin@test.com',
    N'pending@test.com',
    N'banned@test.com'
)
ORDER BY
    CASE r.[roleName]
        WHEN N'Customer' THEN 1
        WHEN N'Staff' THEN 2
        WHEN N'Manager' THEN 3
        WHEN N'Admin' THEN 4
        ELSE 5
    END,
    u.[email];
GO

SELECT
    c.[cinemaId],
    r.[roomId],
    m.[movieId],
    s.[showtimeId],
    COUNT(ss.[showtimeSeatId]) AS [showtimeSeatCount]
FROM [CINEMA] c
JOIN [ROOM] r ON r.[cinemaId] = c.[cinemaId]
JOIN [SHOWTIME] s ON s.[roomId] = r.[roomId]
JOIN [MOVIE] m ON m.[movieId] = s.[movieId]
LEFT JOIN [SHOWTIME_SEAT] ss ON ss.[showtimeId] = s.[showtimeId]
WHERE c.[cinemaId] = N'CIN_TEST_ALL_ROLES'
GROUP BY c.[cinemaId], r.[roomId], m.[movieId], s.[showtimeId];
GO
