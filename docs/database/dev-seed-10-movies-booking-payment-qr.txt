rows and never changes/deletes an existing booking, payment, ticket or seat.

-- Pick ST_TEST_QR_01 for the immediate booking/payment/QR test.
SELECT
    showtime.[showtimeId], movie.[title], showtime.[roomId], room.[cinemaId],
    showtime.[startTime], showtime.[endTime], showtime.[basePrice],
    COUNT(seat.[showtimeSeatId]) AS [availableSeatCount]
FROM dbo.[SHOWTIME] AS showtime
INNER JOIN dbo.[MOVIE] AS movie ON movie.[movieId] = showtime.[movieId]
INNER JOIN dbo.[ROOM] AS room ON room.[roomId] = showtime.[roomId]
LEFT JOIN dbo.[SHOWTIME_SEAT] AS seat
    ON seat.[showtimeId] = showtime.[showtimeId] AND seat.[seatStatus] = 'AVAILABLE'
WHERE showtime.[showtimeId] LIKE 'ST_TEST_QR_%'
GROUP BY showtime.[showtimeId], movie.[title], showtime.[roomId], room.[cinemaId],
         showtime.[startTime], showtime.[endTime], showtime.[basePrice]
ORDER BY showtime.[startTime];

-- Find a Staff account allowed to scan the immediate test ticket.
SELECT userAccount.[email], userAccount.[fullName], staff.[staffProfileId], staff.[cinemaId]
FROM dbo.[USER] AS userAccount
INNER JOIN dbo.[ROLE] AS role ON role.[roleId] = userAccount.[roleId]
INNER JOIN dbo.[STAFF_PROFILE] AS staff ON staff.[userId] = userAccount.[userId]
WHERE role.[roleName] = 'STAFF'
  AND userAccount.[status] = 'ACTIVE'
  AND staff.[employmentStatus] = 'ACTIVE'
  AND staff.[cinemaId] = 'CIN_ND_Q1';
GO
